# üìÅ FILE & DIRECTORY MANAGEMENT
ls                # List files & folders
ls -l             # Long listing (permissions, owner, size, date)
ls -a             # Show hidden files


pwd               # Print working directory
cd myfolder/      # Change directory
cd ..             # Go up one directory
cd ~              # Go to home directory



mkdir newfolder   # Make new directory
rmdir myfolder    # Remove empty directory
rm file.txt       # Remove file
rm -r myfolder/   # Remove directory and its contents



cp file1.txt file2.txt       # Copy file
cp -r dir1/ dir2/            # Copy directory recursively
mv file1.txt file2.txt       # Rename or move file/folder
touch newfile.txt            # Create empty file



cat file.txt      # Show file contents
less file.txt     # View file with scroll
more file.txt     # View file page by page
head file.txt     # First 10 lines
tail file.txt     # Last 10 lines
tail -f logfile   # Follow file output in real-time (logs)

grep "text" file.txt  # Search text in file
grep -i "text" file   # Case-insensitive search
grep -r "text" dir/   # Recursively search inside dir
